Create S, P, J, SPJ tables as specified below and insert a few rows in each table:-
create table S
( 	sno varchar(10),
	sname char(15),
	status int(4),
	city char(15),
	primary key(sno)	
);
insert into S
values('s1','ram',20,'london'),
	('s2','shyam',10,'athens'),
	('s3','kedar',20,'london'),
	('s4','virat',30,'athens');

create table P
( 	pno varchar(10),
	pname char(15),
	color char(10),
	weight float(7,2),
	city char(15),
	primary key(pno)	
);
insert into P
values('p1','tube','black',11,'london'),
	('p2','remote','silver',13,'athens'),
	('p3','display','white',12,'london'),
	('p4','shield','black',14,'athens');

create table J
( 	jno varchar(10),
	jname char(15),
	city char(15),
	primary key(jno)	
);
insert into J
values('j1','assembly','london'),
	('j2','processing','athens'),
	('j3','demo','athens'),
	('j4','modeling','athens');

create table SPJ
( 	sno varchar(10),
	pno varchar(10),
	jno varchar(10),
	qty char(10),
	Foreign Key(sno) references s(sno),
	Foreign Key(pno) references p(pno),
	Foreign Key(jno) references j(jno)		
);
